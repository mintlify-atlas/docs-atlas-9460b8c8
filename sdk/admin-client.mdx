## Overview

The `AdminClient` provides full administrative access to manage all aspects of your Weapon-X deployment, including projects, configurations, rules, roles, authentication tokens, and environments.

## Initialization

```typescript
import { AdminClient } from 'weapon-x-sdk';

const admin = new AdminClient({
  baseUrl: 'https://api.weapon-x.io',
  headers: {
    Authorization: 'Bearer your-admin-api-key'
  }
});
```

**Request Parameters:**

```typescript
interface CreateConfigRequest {
  project_reference: string;           // Project identifier
  key: string;                         // Unique config key
  description?: string;                // Human-readable description
  type: 'BOOLEAN' | 'JSON' | 'STRING' | 'SECRET';
  is_active: boolean;                  // Enable/disable config
  default_value: unknown;              // Default value when no rules match
  validation_schema?: Record<string, unknown>; // JSON schema for validation
}
```

### assignRule

Add a conditional rule to an existing configuration.

```typescript
async assignRule(configKey: string, data: CreateRuleRequest): Promise<Rule>
```

<Tabs>
  <Tab title="Simple Condition">
    ```typescript
    const rule = await admin.assignRule('enable-new-ui', {
      project_reference: 'my-app',
      name: 'Enable for beta users',
      return_value: true,
      conditions: [
        {
          attribute: 'user_type',
          operator: 'EQUALS',
          value: 'beta'
        }
      ]
    });
    ```
  </Tab>
  
  <Tab title="Multiple Conditions">
    ```typescript
    const rule = await admin.assignRule('premium-features', {
      project_reference: 'my-app',
      name: 'Premium users in US',
      return_value: true,
      conditions: [
        {
          attribute: 'subscription',
          operator: 'EQUALS',
          value: 'premium'
        },
        {
          attribute: 'country',
          operator: 'IN',
          value: ['US', 'CA', 'UK']
        }
      ]
    });
    ```
  </Tab>
  
  <Tab title="With Rollout">
    ```typescript
    const rule = await admin.assignRule('new-algorithm', {
      project_reference: 'my-app',
      name: 'Gradual rollout to all users',
      return_value: true,
      conditions: [], // Empty = matches all
      rollout_percentage: 25 // Only 25% of users
    });
    ```
  </Tab>
</Tabs>

**Request Parameters:**

```typescript
interface CreateRuleRequest {
  project_reference: string;
  name: string;
  return_value: unknown;
  conditions: Condition[];
  rollout_percentage?: number;  // 0-100, defaults to 100
}

interface Condition {
  attribute: string;                    // Context attribute to check
  operator: Operator;                   // Comparison operator
  value: string | string[];             // Value(s) to compare against
}

type Operator =
  | 'EQUALS'
  | 'NOT_EQUALS'
  | 'IN'
  | 'NOT_IN'
  | 'CONTAINS'
  | 'GREATER_THAN'
  | 'LESS_THAN'
  | 'REGEX';
```



## Complete Example

```typescript
import { AdminClient } from 'weapon-x-sdk';

const admin = new AdminClient({
  baseUrl: 'https://api.weapon-x.io',
  headers: {
    Authorization: 'Bearer admin-key'
  }
});

// 1. Create a project
const project = await admin.createProject({
  reference: 'ecommerce',
  name: 'E-Commerce Platform'
});

// 2. Create a role
const role = await admin.createRole({
  name: 'App Service',
  permissions: ['read:configs', 'evaluate:configs']
});

// 3. Create authentication
const auth = await admin.createAuthentication({
  project_reference: 'ecommerce',
  role_id: role.id,
  secret_key: 'secure-generated-key',
  email: 'service@ecommerce.com',
  description: 'Backend service credentials'
});

// 4. Create a feature flag
const flag = await admin.createConfig({
  project_reference: 'ecommerce',
  key: 'checkout-v2',
  description: 'New checkout flow',
  type: 'BOOLEAN',
  is_active: true,
  default_value: false
});

// 5. Add a rule for premium users
const rule = await admin.assignRule('checkout-v2', {
  project_reference: 'ecommerce',
  name: 'Premium users only',
  return_value: true,
  conditions: [
    {
      attribute: 'subscription_tier',
      operator: 'EQUALS',
      value: 'premium'
    }
  ],
  rollout_percentage: 50 // 50% gradual rollout
});

console.log('Setup complete!');
```

## Type Definitions

### Config

```typescript
interface Config {
  id: string;
  project_reference: string;
  key: string;
  description: string;
  type: 'BOOLEAN' | 'JSON' | 'STRING' | 'SECRET';
  is_active: boolean;
  default_value: unknown;
  validation_schema: Record<string, unknown>;
  created_at: string;
  updated_at: string;
  rules?: Rule[];
}
```

### Rule

```typescript
interface Rule {
  id: string;
  configuration_id: string;
  name: string;
  conditions: Condition[];
  return_value: unknown;
  priority: number;
  rollout_percentage: number;
  created_at: string;
  updated_at: string;
}
```

### Project

```typescript
interface Project {
  reference: string;
  name: string;
  created_at: string;
  updated_at: string;
  configurations?: Config[];
  authentications?: Authentication[];
}
```

### Role

```typescript
interface Role {
  id: string;
  name: string;
  permissions: string[];
  created_at: string;
  updated_at: string;
}
```

### Authentication

```typescript
interface Authentication {
  id: string;
  project_reference: string;
  role_id: string;
  secret_key: string;
  email: string;
  description: string;
  is_active: boolean;
  expiration_date: string | null;
  created_at: string;
  updated_at: string;
  removed_at: string | null;
  role?: Role;
}
```
